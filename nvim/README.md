```bash
├── lua
│  ├── config # keybindings, options, …
│  ├── funcs # utility functions & personal plugins
│  └── plugins # plugins & their configs
├── after
│  └── ftplugin # filetype-specific configs
├── queries # modifications of treesitter queries
├── snippets
│  ├── basic
│  └── project-specific
└── templates
```
